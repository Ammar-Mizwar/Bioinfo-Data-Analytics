# Change to 1D using flatten
O = N.flatten()
O[1] = 10  # <- Change second element to 10
print("\nModified flattened COPY:\n\n", O, "\n")
print("Original N after .flatten():\n\n", N, "\n")

# Change to 1D using ravel
P = N.ravel()
P[1] = 10  # <- Change second element to 10
print("\nModified flattened COPY:\n\n", P, "\n")
print("Original N after .ravel():\n\n", N, "\n")

# Explanation
print("\nExplanation:\n")
print("The flatten() method returns a new, independent 1D copy of the original array.\n")
print("Any modifications to the flattened copy do not affect the original array.\n")
print("On the other hand, ravel() returns a flattened view (not a copy).\n")
print("Meaning any changes made to the raveled version will directly reflect in the original array.\n")

import copy

dictionary = {'Sophia': [97, 88]}     # Original dictionary

#-----SHALLOW COPY-----
shallow1 = dict(dictionary)           # Shallow Copy using dict() constructor
shallow2 = dictionary.copy()          # Shallow Copy using copy() method of dict
dictionary['Sophia'][0] = 100         # Modify the inner list of the original

# Display results after shallow copy change
print("Original dictionary:\n", dictionary)
print("Shallow copy 1 (dict()):\n", shallow1)
print("Shallow copy 2 (.copy()):\n", shallow2)

#-----DEEP COPY-----
deep_copy = copy.deepcopy(dictionary) # Using deepcopy from the copy module
dictionary['Sophia'][1] = 50          # Modify original again

# Display after deep copy to prove separation
print("\nAfter deepcopy + second modification:")
print("Original dictionary:\n", dictionary)
print("Deep copy:\n", deep_copy)

#Showing each memory address
print("\nMemory addresses of inner lists:")
print("Original:", id(dictionary['Sophia']))
print("Shallow1:", id(shallow1['Sophia']))
print("Shallow2:", id(shallow2['Sophia']))
print("DeepCopy:", id(deep_copy['Sophia']))
